正整数分解质因数
将一个正整数分解质因数。例如：输入90，打印出90=2*3*3*5
    public class Solution {
        public static void main(String[] args) {
            Scanner sc = new Scanner(System.in);
            System.out.println("请输入要分解的正整数：");
            int n = sc.nextInt();
            StringBuilder sb = new StringBuilder();
            sb.append(n + "=");
            //逻辑是：从2-n中找一个最小的且能被n整除的数k
            //如果找到了 就用 n/k 的商作为新n  继续循环
            //如果n == k 则说明是质数 或者商是质数 就循环结束直接输出
            for (int i = 2; i < n + 1; i++) {
                while (n % i == 0 && n != i) {
                    n = n / i;
                    sb.append(i + "*");
                }

                if (n == i) {     //如果上面的都不能整除说明是个质数
                    sb.append(i);
                    break;
                }
            }
        System.out.println(sb.toString());
    }
}
